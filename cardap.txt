  const section = btn.closest('.categoria-group');
  if (!section) return;
  const postList = section.querySelector('.post-list');
  if (postList) {
    postList.classList.toggle('is-collapsed');
    if (postList.classList.contains('is-collapsed')) {
      btn.innerHTML = 'Ver todos';
      btn.classList.remove('is-expanded');
      // Rola suavemente de volta pro título da seção para evitar que o usuário se perca no topo caso estivesse no final
      const headerWrapper = section.querySelector('.categoria-header-wrapper');
      if (headerWrapper) {
        const y = headerWrapper.getBoundingClientRect().top + window.pageYOffset - 130;
        window.scrollTo({top: y, behavior: 'smooth'});
      }
    } else {
      btn.innerHTML = 'Ver menos';
      btn.classList.add('is-expanded');
    }
  }
};

((() => {

  // --- Módulo: models/Produto.js ---
  class Produto {
    constructor(id, nome, precoBase, tamanhoObj, adicionaisArr, pontoCarne, observacoes, quantidade) {
      this.id = id;
      this.nome = nome;
      this.precoBase = parseFloat(precoBase) || 0;
      this.tamanho = tamanhoObj;
      this.adicionais = adicionaisArr || [];
      this.pontoCarne = pontoCarne || "";
      this.observacoes = observacoes || "";
      this.quantidade = parseInt(quantidade) || 1;
      this.thumb = "";
    }
    calcularSubtotal() {
      let precoTamanho = this.tamanho ? this.tamanho.precoExtra : 0;
      let totalAdicionais = this.adicionais.reduce((soma, adic) => soma + adic.preco, 0);
      return (this.precoBase + precoTamanho + totalAdicionais) * this.quantidade;
    }
  }

  // --- Módulo: models/Carrinho.js ---
  class Carrinho {
    constructor() {
      this.itens = this.carregarDoLocalStorage();
      this.taxaEntregaPadrao = 5;
    }
    salvarNoLocalStorage() {
      try {
        localStorage.setItem("meuCarrinho", JSON.stringify(this.itens));
      } catch (e) {
        console.error("Erro ao salvar carrinho no localStorage:", e);
      }
    }
    carregarDoLocalStorage() {
      try {
        const itensSalvos = localStorage.getItem("meuCarrinho");
        if (!itensSalvos) return [];
        const itensArray = JSON.parse(itensSalvos);
        return itensArray.map((itemData) => {
          const prod = new Produto(
            itemData.id, itemData.nome, itemData.precoBase,
            itemData.tamanho, itemData.adicionais, itemData.pontoCarne,
            itemData.observacoes, itemData.quantidade
          );
          prod.thumb = itemData.thumb || "";
          return prod;
        });
      } catch (e) {
        console.error("Erro ao carregar carrinho do localStorage:", e);
        return [];
      }
    }
    adicionarProduto(produto) { this.itens.push(produto); }
    removerProduto(index) { if (index > -1 && index < this.itens.length) { this.itens.splice(index, 1); } }
    alterarQuantidade(index, novaQuantidade) {
      if (index > -1 && index < this.itens.length) {
        const qtd = parseInt(novaQuantidade);
        if (qtd > 0) { this.itens[index].quantidade = qtd; } else { this.removerProduto(index); }
      }
    }
    esvaziar() { this.itens = []; }
    calcularSubtotal() { return this.itens.reduce((total, item) => total + item.calcularSubtotal(), 0); }
    calcularFrete() { 
      const recebimento = document.getElementById("checkout-recebimento");
      const isRetirada = recebimento && recebimento.value === "Retirada";
      if (isRetirada) return 0;
      return this.calcularSubtotal() >= configLoja.metaFreteGratis ? 0 : this.taxaEntregaPadrao; 
    }
    calcularTotalGeral() { return this.calcularSubtotal() + this.calcularFrete(); }
  }

          priceDisplayEl.textContent = "R$ 0,00";
          return;

        let cleaned = priceLabel.replace(/price_/i, "").replace(/[^0-9.,]/g, "").replace(",", ".");
        let val = parseFloat(cleaned);
        if (isNaN(val)) val = 0;
        priceDisplayEl.textContent = "R$ " + this.formatBRL(val);
      });
    }

        if (container) {

  // --- Módulo: controllers/LojaController.js ---
  class LojaController {
    constructor() {
      this.carrinho = new Carrinho();
      this.view = new LojaView();
      this.cartStep = 1;
      this.init();
    }
    init() {
      const cartEl = document.getElementById('cart');
      if(cartEl) cartEl.classList.add('show-step-1');
      this.bindGlobalEvents();
      this.bindInputEvents();
      this.initScrollSpy();
      this.configurarNavegação();
      this.carregarDadosSalvos();
      this.atualizarCarrinhoCompleto();
    }
    carregarDadosSalvos() {
      const campos = ['checkout-nome', 'checkout-tel', 'checkout-cep', 'checkout-end', 'checkout-bairro'];
      campos.forEach(id => {
        const el = document.getElementById(id);
        const valorSalvo = localStorage.getItem(id);
        if (el && valorSalvo) {
          el.value = valorSalvo;
        }
      });
      
      const selects = ['checkout-recebimento', 'checkout-pagamento'];
      selects.forEach(id => {
        const el = document.getElementById(id);
        const valorSalvo = localStorage.getItem(id);
        if (el && valorSalvo) {
          el.value = valorSalvo;
          el.dispatchEvent(new Event('change', { bubbles: true }));
        }
      });
    }
    navigateToStep(stepNumber) {
      this.cartStep = stepNumber;
      const cartEl = document.getElementById('cart');
      if (!cartEl) return;

      // Remove all step classes
      cartEl.classList.remove('show-step-1', 'show-step-2', 'show-step-3');
      
      // Add the current step class
      cartEl.classList.add(`show-step-${stepNumber}`);

      // Update header
      const titleEl = cartEl.querySelector('#step-title-text') || cartEl.querySelector('.cart-step-title');
      const headerResume = cartEl.querySelector('#header-cart-resume');
      const backBtn = cartEl.querySelector('.btn-voltar');
      
      if (titleEl) {
        if (stepNumber === 1) titleEl.textContent = 'Meu Carrinho';
        if (stepNumber === 2) titleEl.textContent = 'Endereço de Entrega';
        if (stepNumber === 3) titleEl.textContent = 'Pagamento';
        if (stepNumber === 1) {
            titleEl.textContent = 'Meu Carrinho';
            if(headerResume) headerResume.style.display = 'inline-block';
        } else if (stepNumber === 2) {
            titleEl.textContent = 'Endereço de Entrega';
            if(headerResume) headerResume.style.display = 'none';
        } else if (stepNumber === 3) {
            titleEl.textContent = 'Pagamento';
            if(headerResume) headerResume.style.display = 'none';
        }
      }
      if (backBtn) {
        backBtn.style.visibility = (stepNumber > 1) ? 'visible' : 'hidden';
      }
    }
    atualizarCarrinhoCompleto() {
      this.view.renderizarCarrinho(this.carrinho);
      this.view.atualizarFrete(this.carrinho.calcularSubtotal());
      this.carrinho.salvarNoLocalStorage();
    }
    abrirModal(produtoCard) {
      const modal = document.getElementById("modal-global");
      if (!modal || !produtoCard) return;
      modal.produtoAtual = produtoCard;
      modal.querySelector("#modal-title").textContent = produtoCard.querySelector(".item_name").textContent;
      modal.querySelectorAll("select").forEach((s) => s.selectedIndex = 0);
      modal.querySelectorAll(".chk-adicional").forEach((c) => c.checked = false);
      const qtyInput = modal.querySelector(".item_Quantity");
      if (qtyInput) qtyInput.value = 1;
        const obsInput = modal.querySelector(".item_Obs");
        if (obsInput) obsInput.value = "";
      this.atualizarPrecoModal();
      modal.style.display = "flex";
      document.body.style.overflow = "hidden";
    }
    atualizarPrecoModal() {
      const modal = document.getElementById("modal-global");
      if (!modal || !modal.produtoAtual) return;
      const priceDisplay = modal.produtoAtual.querySelector(".price-display");
      const basePriceText = priceDisplay ? priceDisplay.textContent.replace(/[^0-9,]/g, "").replace(",", ".") : "0";
      const precoBase = parseFloat(basePriceText);
      let extra = 0;
      const selTamanho = modal.querySelector(".select_Tamanho");
      if (selTamanho && selTamanho.value) extra += parseFloat(selTamanho.value.split("|")[1]) || 0;
      modal.querySelectorAll(".chk-adicional:checked").forEach((chk) => {
        extra += parseFloat(chk.value.split("|")[1]) || 0;
      });
      const qtyInput = modal.querySelector(".item_Quantity");
      const quantidade = qtyInput ? parseInt(qtyInput.value) || 1 : 1;
      const finalPrice = (precoBase + extra) * quantidade;
      modal.querySelector(".modal-preco").textContent = this.view.formatBRL(finalPrice);
    }
    adicionarDoModal() {
      const modal = document.getElementById("modal-global");
      const produtoCard = modal.produtoAtual;
      if (!produtoCard) return;
      const nome = produtoCard.querySelector(".item_name").textContent;
      const priceDisplay = produtoCard.querySelector(".price-display");
      const precoBase = parseFloat(priceDisplay.textContent.replace(/[^0-9,]/g, "").replace(",", "."));
      const img = produtoCard.querySelector(".item_thumb")?.src || "";
      const selTamanho = modal.querySelector(".select_Tamanho");
      let tamanhoObj = null;
      if (selTamanho && selTamanho.value) {
        const parts = selTamanho.value.split("|");
        tamanhoObj = { nome: parts[0], precoExtra: parseFloat(parts[1]) || 0 };
      }
      const selPonto = modal.querySelector(".select_Ponto");
      const pontoCarne = selPonto ? selPonto.value.split("|")[0] : "";
      const adicionaisArr = [];
      modal.querySelectorAll(".chk-adicional:checked").forEach((chk) => {
        const parts = chk.value.split("|");
        adicionaisArr.push({ nome: parts[0], preco: parseFloat(parts[1]) || 0 });
      });
      const qtyInput = modal.querySelector(".item_Quantity");
      const quantidade = qtyInput ? parseInt(qtyInput.value) || 1 : 1;
        
        const obsInput = modal.querySelector(".item_Obs");
        const observacoes = obsInput ? obsInput.value.trim() : "";
        
        const prod = new Produto(Date.now(), nome, precoBase, tamanhoObj, adicionaisArr, pontoCarne, observacoes, quantidade);
      prod.thumb = img;
      this.carrinho.adicionarProduto(prod);
      this.atualizarCarrinhoCompleto();
      this.view.mostrarToast(nome);
      this.view.animarCarrinho();
      modal.style.display = "none";
      document.body.style.overflow = "";
    }
    bindGlobalEvents() {
      document.addEventListener("click", (e) => {
        const produtoCard = e.target.closest(".btn-abre-modal");
        if (produtoCard) {
          e.preventDefault();
          this.abrirModal(produtoCard);
          return;
        }
        const addBtn = e.target.closest(".btn-ok-modal");
        if (addBtn) {
          this.adicionarDoModal();
          return;
        }
        const closeModal = e.target.closest(".close-modal") || (e.target.matches(".modal-adicionais") ? e.target : null);
        if (closeModal) {
          closeModal.closest(".modal-adicionais").style.display = "none";
          document.body.style.overflow = "";
          return;
        }
        const btnGoToAddress = e.target.closest('#btn-goto-address');
        if (btnGoToAddress) {
          e.preventDefault();
          if (this.carrinho.itens.length === 0) {
            alert("Seu carrinho está vazio!");
            return;
          }
          this.navigateToStep(2);
          return;
        }
        const btnGoToPayment = e.target.closest('#btn-goto-payment');
        if (btnGoToPayment) {
          e.preventDefault();
          
          // Validação aprimorada dos campos do Passo 2
          const nome = document.getElementById("checkout-nome");
          const tel = document.getElementById("checkout-tel");
          const end = document.getElementById("checkout-end");
          const bairro = document.getElementById("checkout-bairro");
          const recebimento = document.getElementById("checkout-recebimento").value;
          
          document.querySelectorAll('#step-2 .input-error').forEach(el => el.classList.remove('input-error'));
          let camposComErro = [];

          if (!nome.value.trim()) { camposComErro.push(nome); }
          
          const telValue = tel.value.replace(/\D/g, '');
          if (!tel.value.trim() || telValue.length < 11) {
              camposComErro.push(tel);
          }
          
          if (recebimento === 'Entrega') {
              if (!end.value.trim()) { camposComErro.push(end); }
              if (!bairro.value.trim()) { camposComErro.push(bairro); }
          }
          
          if (camposComErro.length > 0) {
              camposComErro.forEach(campo => {
                  campo.classList.remove('shake-anim');
                  void campo.offsetWidth; // Força o reflow para reiniciar a animação
                  campo.classList.add('input-error', 'shake-anim');
              });
              camposComErro[0].focus();
              const msg = camposComErro.includes(tel) && telValue.length > 0 && telValue.length < 11 ? "O número de telefone parece inválido. Por favor, insira um número completo com DDD (11 dígitos)." : "Por favor, preencha os campos obrigatórios destacados em vermelho para continuar.";
              alert(msg);
              return;
          }

          // Aciona a captura de GPS automaticamente ao avançar para pagamento
          if (recebimento === 'Entrega') {
              const btnGps = document.getElementById('btn-gps');
              btnGoToPayment.disabled = false;
              if (btnGps && !end.value.includes('GPS:')) {
                  btnGps.click();
              }
          }

          // Garante que o estado do troco inicie correto (oculto se não for dinheiro)
          const pgto = document.getElementById("checkout-pagamento");
          if (pgto) pgto.dispatchEvent(new Event('change', { bubbles: true }));

          this.navigateToStep(3);
          return;
        }
      const openCart = e.target.closest(".viz") || e.target.closest(".carrinho-info") || e.target.closest("#btn-ver-sacola-rodape");
        if (openCart && !document.querySelector(".cartzap").classList.contains("open-cart")) {
          document.querySelector(".cartzap").classList.add("open-cart");          
          document.body.style.overflow = "hidden";          
          return;
        }
        const closeCart = e.target.closest("#close-cart") || e.target.closest(".btn-add-more");
        if (closeCart) {
          document.querySelector(".cartzap").classList.remove("open-cart");
          document.body.style.overflow = "";
          this.navigateToStep(1);
          return;
        }
        const btnBack = e.target.closest('.btn-voltar');
        if (btnBack) {
          e.preventDefault();
          if (this.cartStep > 1) this.navigateToStep(this.cartStep - 1);
          return;
        }
        const cartInc = e.target.closest(".cart-inc");
        if (cartInc) {
          e.preventDefault();
          const index = cartInc.dataset.index;
          const item = this.carrinho.itens[index];
          if (item) this.carrinho.alterarQuantidade(index, item.quantidade + 1);
          this.atualizarCarrinhoCompleto();
          return;
        }
        const cartDec = e.target.closest(".cart-dec");
        if (cartDec) {
          e.preventDefault();
          const index = cartDec.dataset.index;
          const item = this.carrinho.itens[index];
          if (item) this.carrinho.alterarQuantidade(index, item.quantidade - 1);
          this.atualizarCarrinhoCompleto();
          return;
        }
        const cartRem = e.target.closest(".cart-rem");
        if (cartRem) {
          e.preventDefault();
          const itemRow = cartRem.closest('.itemRow');

   initScrollSpy() {
      let lastScrollTop = 0;
      const navFixa = document.getElementById("nav-fixa");
      const pills = document.querySelectorAll(".cat-pill");
      const sections = Array.from(pills).map((pill) => {
        try {
          const selector = pill.getAttribute("href");
          return document.querySelector(selector);
        } catch(e) { return null; }
      }).filter(Boolean);

      window.addEventListener("scroll", () => {
        const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
        
        const categoriasContainer = document.querySelector(".categorias-container");
        const apresentacao = document.querySelector(".apresentacao-inicial");
        const headerLoja = document.getElementById("header-loja");
        
        // CONTROLE DE FIXAÇÃO COM COMPENSAÇÃO DE ALTURA (ANTI-GLITCH)
        if (categoriasContainer && apresentacao && headerLoja) {
          const triggerPoint = apresentacao.offsetHeight;
          if (scrollTop >= triggerPoint) {
            categoriasContainer.classList.add("sombra-fixa");
            // Preenche o buraco deixado pela barra com padding
            headerLoja.style.paddingBottom = categoriasContainer.offsetHeight + "px";
          } else {
            categoriasContainer.classList.remove("sombra-fixa");
            // Remove o padding quando volta para o topo
            headerLoja.style.paddingBottom = "0px";
          }
        }
        
        if (navFixa) {
          if (scrollTop > 250 && scrollTop < lastScrollTop) {
            navFixa.classList.add("nav-fixa-ativa");
          } else {
            navFixa.classList.remove("nav-fixa-ativa");
          }
        }
        
        let currentSectionId = "";
        for (const section of sections) {
            if (section) {
                const sectionTop = section.offsetTop;
                if (scrollTop >= sectionTop - 150) {
                    currentSectionId = section.id;
                }
            }
        }
        
        pills.forEach((pill) => {
          const isActive = pill.getAttribute("href") === `#${currentSectionId}`;
          pill.classList.toggle("active", isActive);
          if (isActive && categoriasContainer) {
            const scrollArea = document.querySelector(".categorias-scroll");
            if (scrollArea) {
              const pos = pill.offsetLeft - (scrollArea.offsetWidth / 2) + (pill.offsetWidth / 2);
              scrollArea.scrollTo({ left: pos, behavior: 'smooth' });
            }
          }
        });
        
        lastScrollTop = scrollTop <= 0 ? 0 : scrollTop;
      }, { passive: true });
    }
    configurarNavegação() {
        document.querySelectorAll('.cat-pill').forEach(pill => {
            pill.addEventListener('click', (e) => {
                e.preventDefault();
                const targetId = pill.getAttribute('href');
                if (!targetId || targetId === '#') return;

                const secao = document.querySelector(targetId);
                if (secao) {
                    // O valor '120' pode ser ajustado dependendo da altura do seu cabeçalho fixo
                    window.scrollTo({ top: secao.offsetTop - 120, behavior: 'smooth' });
                }
            });
        });
    }
    handlePhoneInput(telInput) {
        let v = telInput.value.replace(/\D/g, '');
        v = v.substring(0, 11);
        if (v.length > 2) v = v.replace(/^(\d{2})(\d)/g, '($1) $2');
        if (v.length > 9) v = v.replace(/(\d{5})(\d)/, '$1-$2');
        else if (v.length > 8) v = v.replace(/(\d{4})(\d)/, '$1-$2');
        telInput.value = v;
    }
    handleCepInput(cepInput) {
        let v = cepInput.value.replace(/\D/g, '');
        v = v.substring(0, 8);
        if (v.length > 5) {
            v = v.replace(/^(\d{5})(\d)/, '$1-$2');
        }
        cepInput.value = v;

        // Reseta a cor da borda ao digitar
        cepInput.style.borderColor = 'var(--border-input)';

        if (v.length === 9) { // Formato 00000-000 completo
            this.buscarCep(v.replace('-', ''));
        }
    }
    buscarCep(cep) {
        const endInput = document.getElementById("checkout-end");
        const bairroInput = document.getElementById("checkout-bairro");
        const cepInput = document.getElementById("checkout-cep");

        if (!endInput || !bairroInput || !cepInput) return;

        endInput.value = "Buscando endereço...";
        bairroInput.value = "";

        fetch(`https://viacep.com.br/ws/${cep}/json/`)
            .then(response => response.json())
            .then(data => {
                if (data.erro) {
                    endInput.value = "";
                    cepInput.style.borderColor = '#e74c3c'; // Vermelho para erro
                    alert("CEP não encontrado. Por favor, verifique o número digitado.");
                } else {
                    cepInput.style.borderColor = '#2ecc71'; // Verde para sucesso
                    endInput.value = data.logradouro ? data.logradouro + ", " : "";
                    bairroInput.value = data.bairro || "";
                    document.getElementById("checkout-end").focus(); // Move o cursor para o cliente digitar o número da casa
                }
            })
            .catch(error => {
                console.error("Erro ao buscar CEP:", error);
                endInput.value = "";
            });
    }
    processarCheckout() {
        if (this.carrinho.itens.length === 0) {
            alert("Seu carrinho está vazio!");
            return;
        }

        if (this.carrinho.calcularSubtotal() < configLoja.pedidoMinimo) {
            const falta = (configLoja.pedidoMinimo - this.carrinho.calcularSubtotal()).toFixed(2).replace('.', ',');
            alert(`O valor mínimo do pedido é de R$ ${configLoja.pedidoMinimo.toFixed(2).replace('.', ',')}. Faltam R$ ${falta}.`);
            return;
        }

        const nome = document.getElementById("checkout-nome").value.trim();
        const tel = document.getElementById("checkout-tel").value.trim();
        const recebimento = document.getElementById("checkout-recebimento").value;
        const end = document.getElementById("checkout-end").value.trim();
        const bairro = document.getElementById("checkout-bairro").value.trim();

        if (recebimento === 'Entrega' && (!nome || !tel || !end || !bairro)) {
            alert("Por favor, preencha Nome, Telefone, Endereço e Bairro.");
            return;
        } else if (recebimento === 'Retirada' && (!nome || !tel)) {
            alert("Por favor, preencha Nome e Telefone.");
            return;
        }

        if (recebimento === 'Retirada') {
            this.enviarPedidoWhatsApp(null);
        } else {
            const overlay = document.getElementById('gps-overlay');
            const timerEl = document.getElementById('gps-timer');
            let seconds = 0;
            let timerId = null;

            if (overlay) overlay.style.display = 'flex';
            if (timerEl) {
              timerEl.textContent = 'Tempo: 0s';
              timerId = setInterval(() => {
                seconds++;
                timerEl.textContent = `Tempo: ${seconds}s`;
              }, 1000);
            }

            if ("geolocation" in navigator) {
                navigator.geolocation.getCurrentPosition(
                    (position) => {
                        if (timerId) clearInterval(timerId);
                        if (overlay) overlay.style.display = 'none';
                        this.enviarPedidoWhatsApp({ lat: position.coords.latitude, lng: position.coords.longitude });
                    },
                    (error) => {
                        if (timerId) clearInterval(timerId);
                        if (overlay) overlay.style.display = 'none';
                        let errorMsg = "Não foi possível obter a localização. Verifique as permissões do navegador.";
                        if (error.code === 1) errorMsg = "Você negou a permissão de localização. O pedido será enviado com o endereço digitado.";
                        else if (error.code === 2) errorMsg = "Sinal de GPS indisponível. O pedido será enviado com o endereço digitado.";
                        else if (error.code === 3) errorMsg = "A busca por GPS demorou muito. O pedido será enviado com o endereço digitado.";
                        
                        if (window.location.protocol !== 'https:' && /Mobi|Android|iPhone/i.test(navigator.userAgent)) {
                            errorMsg += "\n\n(Lembre-se: O GPS no celular exige que o site esteja com cadeado HTTPS)";
                        }
                        alert(errorMsg);
                        this.enviarPedidoWhatsApp(null);
                    },
                    {
                        enableHighAccuracy: true,
                        timeout: 30000,
                        maximumAge: 10000 // Permite cache recente de 10s para acelerar leitura no celular
                    }
                );
            } else {
                if (timerId) clearInterval(timerId);
                if (overlay) overlay.style.display = 'none';
                alert("Geolocalização não é suportada neste navegador.");
                this.enviarPedidoWhatsApp(null);
            }
        }
    }
    enviarPedidoWhatsApp(coords) {
        const nome = document.getElementById("checkout-nome").value.trim();
        const tel = document.getElementById("checkout-tel").value.trim();
        const recebimento = document.getElementById("checkout-recebimento").value;

        this.carrinho.itens.forEach(item => {

            }
            pedido += `\n`;
        });

        pedido += "--------------------------------------\n*RESUMO FINANCEIRO:*\n";

                pedido += `*Localização GPS:* https://www.google.com/maps?q=${coords.lat},${coords.lng}\n`;
            }
        }
        if (pagamento === 'Dinheiro' && troco) pedido += `*Troco para:* R$ ${troco}\n`;
        if (obs) pedido += `*Observações Gerais:* ${obs}\n`;
        pedido += "\n--------------------------------------\n🙏 _Agradecemos a preferência!_";

        const url = `https://api.whatsapp.com/send?phone=${configLoja.whatsapp}&text=${encodeURIComponent(pedido)}`;
        window.open(url, '_blank');
